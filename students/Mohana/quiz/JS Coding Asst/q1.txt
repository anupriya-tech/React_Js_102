function calGrade(){
  var perc = document.getElementById('percent').value;
if(perc >=90  && perc <=100){
    alert('A');
  }
  else if(perc >80  && perc < 90){
     alert('B');
  }else if(perc >=70  && perc <=80){
    alert('C');
  }else{
     alert('FAIL');
  }
}
