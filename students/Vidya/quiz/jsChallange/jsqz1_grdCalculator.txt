<script lang="Javascript">
function calculate()
{
	var stdName = document.getElementById("inputStdName").value
	var sub1 = parseInt(document.getElementById("inputSub1").value);
	var sub2 = parseInt(document.getElementById("inputSub2").value);
	var sub3 = parseInt(document.getElementById("inputSub3").value);
	var sub4 = parseInt(document.getElementById("inputSub4").value);

	var totalMarks = sub1 + sub2 + sub3 + sub4;
	

	var per = (totalMarks / 400) * 100 ;

	if(per >= 90)
	{
	 document.write("Name: " + stdName + "<br> Total : "+ totalMarks + "/400 <br> Perecentage : "+ per +" <br> You are Passed with Grade A");
	}
	else if(per >= 80 && per < 90)
	{
	 document.write("Name: " + stdName + "<br> Total : "+ totalMarks + "/400 <br> Perecentage : "+ per +" <br> You are Passed with Grade B");
	}
	else if(per >= 70 && per <= 80)
	{
	 document.write("Name: " + stdName + "<br> Total : "+ totalMarks + "/400 <br> Perecentage : "+ per +" <br> You are Passed with Grade C");
	}
	else
	{
	 document.write("Name: " + stdName + "<br> Total : "+ totalMarks + "/400 <br> Perecentage : "+ per +" <br> You are Fail");
	}

}
</script>
