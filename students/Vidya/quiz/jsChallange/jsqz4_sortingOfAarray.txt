<h2>Sort Records</h2>

<script>
var i;

var std = [	{"id":101,"name":"Raj"},
		   	{"id":121,"name":"Rajesh"},
			{"id":130,"name":"Suraj"},
			{"id":114,"name":"Atom"} ];

 // Function Code For Sorting
function GetSort(prop) {    
    return function(a, b) {    
        if (a[prop] > b[prop]) {    
            return 1;    
        } else if (a[prop] < b[prop]) {    
            return -1;    
        }    
        return 0;    
    }    
}    

document.write("UnSorted Students' Records: ");    
for (i in std) {    
    document.write("<br>" + std[i].id + " " + std[i].name);    
}

std.sort(GetSort("id")); //Pass the attribute to be sorted on : By ID  
document.write("<br><br>Sorted Students' Records By Id : ");    
for (i in std) {    
    document.write("<br>" + std[i].id + " " + std[i].name);    
}    
    
std.sort(GetSort("name")); //Pass the attribute to be sorted on : By Name  
document.write("<br><br> Sorted  Students' Records By Name : ");    
for (i in std) {    
    document.write("<br>" + std[i].id + " " + std[i].name);    
} 


</script>
