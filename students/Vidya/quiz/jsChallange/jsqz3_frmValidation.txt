<script lang="Javascript">
function validate()
{
	var userName = document.getElementById("inputUserName").value;
	var userPass = document.getElementById("inputUserPass").value;
	var patternForEmail = /^[_a-zA-Z0-9-]+(\.[_a-zA-Z0-9-]+)*@[a-zA-Z0-9-]+(\.[a-zA-Z0-9-]+)*\.(([0-9]{1,3})|([a-zA-Z]{2,4})|(aero|coop|info|museum|name))$/;
	var patternForPassword = /(?=.*[a-z])(?=.*[A-Z]).{6,}/; 
	var resultForEmail = patternForEmail.test(userName);
	var resultForPassword = patternForPassword.test(userPass);
	alert(resultForEmail)
	alert(resultForPassword);
		
	if(userName !== "" && userPass !== "")
	{	
		if(resultForEmail == true)
		{
			return (true);
		}
		else{
			alert("Enter Valid Email Id!");
			return (false)
		}
		
		if(resultForPassword == true)
		{
			return (true)
		}
		else{
			alert("Password can not be less than 6 character and at least one capital letter.!");
			return (false)
		}
	}
	else{
	alert("Fields should not be Empty !");
	}
}
</script>
