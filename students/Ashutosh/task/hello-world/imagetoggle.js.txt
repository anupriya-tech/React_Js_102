import React from 'react';
import ReactDOM from 'react-dom';
const imagesPath = {
  bulbon: "images/bulbon.jpg",
  bulboff: "images/bulboff.jpg"
}
class MyToggleComponent extends React.Component{
  constructor(props){
    super(props);
    this.state={isOn:true,imagePath:"images/bulboff.jpg"};
    this.myHandler=this.myHandler.bind(this);
  }
  myHandler(){
    this.setState(state=>(
    {
      isOn:!state.isOn
      
    }
    ));
    console.log(this.state.isOn +" "+this.getImageName());
  }
  getImageName = () => !this.state.isOn ? 'bulbon' : 'bulboff'
  render(){
    const imageName = this.getImageName();
    
    return(
      <div>
     <button onClick={this.myHandler}>
     {this.state.isOn?'ON':'OFF'}
       </button>
       <br/>
       <img src={imagesPath[imageName]}/>
       
       </div>
    )
  }
}
ReactDOM.render(
  <MyToggleComponent />
  ,document.getElementById("root") );
