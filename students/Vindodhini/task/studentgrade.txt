function calcgrade(){


  var studentname = document.getElementById('studentname').value;
  var maths = document.getElementById('maths').value;
  var grades = '';

  if(maths <= 100  && maths >= 80){
   console.log( grades = 'A');
  }else if(maths <= 79  && maths >= 60){
      console.log( grades = 'B');
  }else if(maths <= 59  && maths >= 40){
     console.log( grades = 'C');
  }else{
      console.log( grades = 'F');
  }

document.write("Student Name : " + studentname + '</br>');
document.write("Grade : " + grades);
}calcgrade;
