2)Quiz_JS_102
var student = {
	name:"John",
	getName: function(){console.log("student"+this.name);}
}

var employee = {
	name:"Tom",
	getName: function(){console.log("employee"+this.name);}
}

var run = student.getName.bind(employee);
run();

Output:studentTom

Description: Here, student.getName is being bind with employee because of which the name of the employee is being displayed instead of student name.

//What does a filter do?

function evenodd(n){
	return n%2 == 0;
}

myarray = [1,2,3,4,5];

console.log(myarray.filter(evenodd));

Output:[1, 3, 5]

Description: The sign given is not equal to. So those array elements which are odd numbers will be returned by the function.

}
